% -------------------------

\usepackage[a4paper,margin=1in]{geometry}

% -------------------------
% Fonts & Math
% -------------------------
\usepackage{amsmath,amssymb}

% -------------------------
% Images & Floats
% -------------------------
\usepackage{graphicx}

% -------------------------
% Code Listings (Python)
% -------------------------
\usepackage{listings}
\usepackage{xcolor}

\renewcommand{\lstlistingname}{Listing}
\renewcommand{\lstlistlistingname}{List of Listings}

\lstdefinestyle{pythonstyle}{
    language=Python,
    basicstyle=\ttfamily\small,
    keywordstyle=\color{blue},
    stringstyle=\color{teal},
    commentstyle=\color{gray},
    numbers=left,
    numberstyle=\tiny\color{gray},
    stepnumber=1,
    frame=single,
    breaklines=true,
    tabsize=4,
    showstringspaces=false,
    captionpos=b
}

% -------------------------
% Hyperlinks
% -------------------------
\usepackage[
    colorlinks=true,
    linkcolor=blue,
    citecolor=green,
    urlcolor=cyan
]{hyperref}

% -------------------------
% Header & Footer Styling
% -------------------------
\usepackage{fancyhdr}
\pagestyle{fancy}

% Clear default settings
\fancyhf{}

% Header
\fancyhead[LE,RO]{\thepage}                 % Page number
\fancyhead[RE]{\nouppercase{\leftmark}}     % Chapter name (even pages)
\fancyhead[LO]{\nouppercase{\rightmark}}    % Section name (odd pages)

% Footer
\fancyfoot[C]{%
  \makebox[\textwidth]{%
    \hrulefill\quad[\thepage]\quad\hrulefill
  }%
}

% Header rule
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0pt}

% -------------------------
% Plain Page Style (Chapter First Page)
% -------------------------
\fancypagestyle{plain}{
    \fancyhf{}
    \fancyhead[LE,RO]{\thepage}
    \fancyhead[RE]{\nouppercase{\leftmark}}
    \fancyhead[LO]{\nouppercase{\rightmark}}
    \fancyfoot[C]{%
      \makebox[\textwidth]{%
        \hrulefill\quad[\thepage]\quad\hrulefill
      }%
    }
    \renewcommand{\headrulewidth}{0.4pt}
}
