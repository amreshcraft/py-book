\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}  % clear all

% HEADER
\fancyhead[LE,RO]{\thepage}                  % page number left (even) / right (odd)
\fancyhead[RE]{\nouppercase{\leftmark}}      % chapter name on even pages
\fancyhead[LO]{\nouppercase{\rightmark}}     % section name on odd pages

% FOOTER
\fancyfoot[C]{[\thepage]}                     % centered page number in brackets

% LINES
\renewcommand{\headrulewidth}{0.4pt}         % thin line above header
\renewcommand{\footrulewidth}{0.2pt}         % thin line above footer
